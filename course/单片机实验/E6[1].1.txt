#include<reg52.h>
#include<string.h>

#define LCD_DB P0
sbit  LCD_RW = P2^5;
sbit  LCD_RS = P2^6;
sbit  LCD_E  = P2^7;

#define uchar unsigned char
#define uint unsigned int

sbit  K1=P3^2;
sbit  K2=P3^3;
uchar count1,count2,time,t1,t2,x1,x2,y1,y2,z1,z2;
unsigned char Disp[] ={0x30,0x31,0x32,0x33,0x34,0x35,0x36,0x37,0x38,0x39};

void LCD_init(void);
void LCD_check_busy(void);
void LCD_write_command(uchar command);
void LCD_write_data(uchar dat);
void delay(uint n);

void int_init();

void T_init();




void LCD_init(void)
{
   LCD_write_command(0x38);
   LCD_write_command(0x0c);
   LCD_write_command(0x06);
   LCD_write_command(0x01);
}


void LCD_check_busy(void)           
{
    unsigned char sta;      
    LCD_DB = 0xff;
    LCD_RS = 0;
    LCD_RW = 1;
    do
    {
        LCD_E= 1;
        sta = LCD_DB;
        LCD_E= 0;    
    }while(sta & 0x80);
}



void LCD_write_command(uchar dat)
{
   LCD_check_busy();
   LCD_DB=dat;
   LCD_RS=0;
   LCD_RW=0;
   LCD_E=1;
   LCD_E=0;
   delay(1);
}


void LCD_write_data(uchar dat)
{
   LCD_check_busy();
   LCD_DB=dat;
   LCD_RS=1;
   LCD_RW=0;
   LCD_E=1;
   LCD_E=0;
   delay(1);
}



void LCD_show_str(char y, char k, char str[])  
{

	uchar i,j, address ;
	if(y==1)
	{
	   j=0x80;
	}
	else j=0xc0;
	for(i = 0;i<strlen(str);i++)
	{
	   address=i+j+k;
	   LCD_write_command(address);
	   LCD_write_data(str[i]);
	}
}

void LCD_show_var(char y,char z, char v1,v2)   

	uchar j, address ;
	if(y==1)
	{
	   j=0x80;
	}
   else j=0xc0;
	address=7*z+5+j;
	LCD_write_command(address);
	LCD_write_data(v1);
	address=7*z+6+j;
	LCD_write_command(address);
   LCD_write_data(v2);
}

//延时函数
void delay(uint n)
{
   uint i;
   uchar j;
   for(i=n;i>0;i--)
   for(j=0;j<2;j++);
}


void Key_int0() interrupt 0
{
   EX0=0;
   count1++;
   time=0;
   EX0=1;
}


void Key_int1() interrupt 2
{
EX1=0;
   count2++;
   time=0;
   EX1=1;
}

void int_init()
{
   IE=0x85;
   PX0=1;	    
   IT0=1;
   IT1=1;
}

void t_int0() interrupt 1
{
   t1++;
}


void T_init()
{
   TMOD=0x02;    
   EA=1;
   TL0=0xce;
   TH0=0xce;
   ET0=1;
}
//主函数
void main(void)
{
   int_init();
   T_init();
   count1=0;
   count2=0;
   t2=0;
   LCD_init();
   while(1)
   {


      x1=count1/10;
      x2=count1%10;
      y1=count2/10;
      y2=count2%10;
      LCD_show_str(1,0,"INT0:");
      LCD_show_str(1,7,"INT1:");
      LCD_show_var(1,0,Disp[x1],Disp[x2]);
      LCD_show_var(1,1,Disp[y1],Disp[y2]);
      TR0=1;
      if(t1>=100)
      {
         t1=0;
         t2++;
      }
      if(t2>=100)
      {
         t2=0;
         time++;
      }
      if(time>=60)
      {
         time=0;
      }
      z1=time/10;
      z2=time%10;
      LCD_show_var(2,0,Disp[z1],Disp[z2]);
   }
}
